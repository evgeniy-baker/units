13)	С помощью вложенных циклов и символа * нарисуйте:

   *
   **
   ***
   **
   *

string = ''
for (let i = 1; i < 4; i++) {
  string += '   '
  for (let k = 0; k < i; k++) {
    string += '*'
  }
  string += '\n'
}
for (let i = 1; i < 4; i++) {
  string += '   '
  for (let k = 3; k > i; k--) {
    string += '*'
  }
  string += '\n'
}
console.log(string)
